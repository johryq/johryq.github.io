Nest 配置 swagger 接口文档#

htu#

  1. 下载所需要的包
npm install  @nestjs/swagger
  1. main.ts注册
import { DocumentBuilder, SwaggerModule } from '@nestjs/swagger';

async function bootstrap() {
  const app = await NestFactory.create<NestExpressApplication>(AppModule);
  const options = new DocumentBuilder().setTitle('接口文档').setDescription('描述，。。。').setVersion('1').build()
  // const options = new DocumentBuilder().addBearerAuth().setTitle('接口文档').setDescription('描述，。。。').setVersion('1').build()
  // 添加 jwt auth
  const document = SwaggerModule.createDocument(app,options)
  SwaggerModule.setup('/api-docs',app,document)
  await app.listen(3000);
}
  1. controller中添加tag标签
// ...controller.ts
import { ApiTags } from '@nestjs/swagger'

@ApiTags('name')
// controller class tag

@ApiOperation({summary:"测试admin",description:"请求该接口需要amdin权限"})
// API description
@ApiParam({name:"id",description:"用户id",required:true})
// API param
@ApiQuery({name:"xxxx",description:"bbb"})
// API query
@ApiResponse({status:403,description:"自定义返回信息"})
// API response
