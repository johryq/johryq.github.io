配置selinux#

# 预设策略

# -P永久打开

---
# 修改文件策略

semanage fcontext -l
# 列出策略

semanage fontext -a -t 
# 添加策略

semanage fcontext -d 
# 删除策略

---
# 端口策略
semanage port -l

semanage port -a -t 策略类型 -p 协议 端口号

semanage port -d
